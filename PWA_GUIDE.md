import json
from django.conf import settings

def send_push_notification(subscription_info, message_body, url='/'):
    """
    –û—Ç–ø—Ä–∞–≤–∏—Ç—å push-—É–≤–µ–¥–æ–º–ª–µ–Ω–∏–µ

    Args:
        subscription_info: dict —Å endpoint, keys (p256dh, auth)

        url: –∫—É–¥–∞ –ø–µ—Ä–µ–π—Ç–∏ –ø—Ä–∏ –∫–ª–∏–∫–µ
    """
    try:
        payload = json.dumps({
            'title': 'Paddle Booking',
            'body': message_body,
            'url': url,
            'icon': '/static/icons/icon-192x192.png'
        })

        webpush(
            subscription_info=subscription_info,
            data=payload,
            vapid_private_key=settings.VAPID_PRIVATE_KEY,
            vapid_claims={
                "sub": f"mailto:{settings.VAPID_EMAIL}"

        )
        return True
    except WebPushException as ex:
        print(f"Push failed: {repr(ex)}")

function urlBase64ToUint8Array(base64String) {
    const padding = '='.repeat((4 - base64String.length % 4) % 4);
    const base64 = (base64String + padding)
        .replace(/\-/g, '+')
        .replace(/_/g, '/');

    const rawData = window.atob(base64);
    const outputArray = new Uint8Array(rawData.length);

    for (let i = 0; i < rawData.length; ++i) {
        outputArray[i] = rawData.charCodeAt(i);
    }
    return outputArray;
